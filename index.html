数据库操作语言

对数据库的操作

创建数据库

CREATE DATABASE  [IF NOT EXISTS] testschool

删除数据库

DROP DATABASE [IF EXISTS] testschool

查询所有数据库

SHOW DATABASES

使用数据库

USE `testschool`

字段属性

Unsigned

无符号整数

声明该列不能为负数

zerofill

0填充

不足位数，0填充

自增

默认+1

高级设置里面可以修改

默认值

非空

设置为NULL，则默认为空

not null 则默认不为空

数据库表操作

增添表

CREATE TABLE IF NOT EXISTS `student`(
`id` INT (10) NOT NULL  AUTO_INCREMENT COMMENT '学号',
`name` VARCHAR(30) NOT NULL DEFAULT '匿名' COMMENT '姓名',
`pwd` VARCHAR (20) NOT NULL DEFAULT '123456' COMMENT '密码',
`sex` VARCHAR(2) NOT NULL DEFAULT '男' COMMENT '性别' ,
`birthday` DATETIME DEFAULT NULL COMMENT '生日',
`address` VARCHAR(100) NOT NULL DEFAULT '保密' COMMENT '地址',
`email` VARCHAR (50) DEFAULT NULL COMMENT '电子邮件地址',
PRIMARY KEY (`id`) 
)ENGINE=INNODB DEFAULT CHARSET=utf8

格式

修改表的字段

ALTER TABLE student RENAME AS student1 -- 命名表名

ALTER TABLE student1 ADD age INT (11) 	-- 新添表的列属性，增加表的字段 ADD 字段名 列属性

ALTER TABLE student1 MODIFY age VARCHAR(11) -- 修改字段的约束

ALTER TABLE student1 CHANGE age age1 INT (11) -- 重命名字段

ALTER TABLE student1 DROP age1 -- 删除表的字段
